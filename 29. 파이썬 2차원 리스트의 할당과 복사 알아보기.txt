#2차원 리스트의 할당과 복사 알아보기

a= [ [10,20] , [30,40] ]
b=a
b[0][0] = 500

a입력시 

[ [500,20] , [30,40] ] 이 출력이 됨

b 입력시
[ [500,20] , [30,40] ] 이 출력이 됨

a= [ [10,20] , [30,40] ]
b=a.copy()
b[0][0] = 500

a입력시 

[ [500,20] , [30,40] ] 이 출력이 됨

b 입력시
[ [500,20] , [30,40] ] 이 출력이 됨

#2차원 이상의 다차원 리스트는 리스트를 완전히 복사할려면
deepcopy 함수를 사용해야함

a= [ [10,20] , [30,40] ]
import copy
b=copy.deepcopy()
b[0][0]=500
a 입력시

[ [10,20] , [30,40] ] 이 출력

b 입력시
[ [500,20] , [30,40] ] 이 출력이 됨
